#include <stdio.h>
#include <string.h>
int romanToInt(char * s){
    int a=0;
	int b=0;
    int sum=0;
    for(i=0;i<strlen(s);i++)
	{
		switch(s[i])
		{
			case 'I': b = 1; break;
            case 'V': b = 5; break;
            case 'X': b = 10; break;
            case 'L': b = 50; break;             
			case 'C': b = 100; break;
            case 'D': b = 500; break;
            case 'M': b = 1000; break;
		}
		if(a<b)
		{
			sum-=a;
		}
		else
		{
			sum+=a;
		}
		a=b;
	}
	sum+=a;
	return sum;
}
